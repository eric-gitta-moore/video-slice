// 导入库
import { videoSlice } from 'video-slice-reversed';

// 创建切片实例
const slicer = videoSlice({
    file: videoFile,           // 视频文件
    width: 640,               // 宽度
    height: 360,              // 高度
    intervalTime: 1000,       // 切片间隔(毫秒)
    muted: true,              // 静音
    
    success: () => {
        console.log('视频加载成功');
    },
    
    backImgFn: (base64) => {
        console.log('获取到切片:', base64);
        // 显示或保存图片
    },
    
    finish: () => {
        console.log('切片完成');
    },
    
    fail: (error) => {
        console.error('失败:', error);
    }
});

// 开始切片
slicer.start();

// 快速切片
slicer.fastStart(500);

// 截取指定时间
slicer.setCurrentTime(10, true);

// 获取所有图片
const images = slicer.getImgs();

// 销毁实例
slicer.destroy();
